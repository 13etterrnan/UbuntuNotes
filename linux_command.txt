#用户管理

		命令				作用		说明
		useradd -m -g组 新建用户名		添加新用户	-m：自动建立用户家目录

#常用Linux命令
	序号	命令		对应英文		作用
	01.	ls		list		查看当前文件夹下的内容
	02.	pwd		print work directory	查看当前所在文件夹
	03.	cd [目录名]	change directory	切换文件夹
	04.	touch [文件名]	touch		如果文件不存在，新建文件
	05.	mkdir [目录名]	make directory	创建目录
	06.	rm [文件名]	remove		删除指定文件名（rm -r xxx 可删除目录）
	07.	clear		clear		清屏

	终端命令格式：command [-options] [parameter]
	说明：
		command：命令名，相应功能的英文你单词或单词的缩写
		[-options]: 选项，可用来对命令进行控制，也可以省略
		[parameter]：传给命令的参数，可以是零个，一个或者多个
	[ ]表示可选

#查阅命令帮助信息
	1.--help：
		command --help
	2.man：
		man command（man是manul缩写，是linux提供的一个手册，包含了绝大部分的命令，函数的详细使用说明）


#命令扩展
	touch：
		如果文件不存在，可以新建一个空白文件
		如果文件已存在，可以修改文件的末次修改日期
	mkdir：
		mkdir -p a/b/c/d/ （-p选项 可以递归创建目录，也就是创建a目录，然后a目录下有b目录...）
		（ps：新建目录名不可与已存在的目录或文件同名）
	rm：
		删除文件或目录（ps：直接从磁盘中干掉，删除后不能恢复）
		rm -r xxx	删除名为xxx的目录
		rm -f xxx	删除名为xxx的文件（加上-f选项后，无论有没有名为xxx的文件将不提示信息）
		（ps：rm -rf * 千万不要在根目录下使用，除非你跟老板有仇；千万不要在根目录下使用，除非你跟老板有仇；千万不要在根目录下使用，除非你跟老板有仇）
	tree：
		tree命令可以以树状图列出文件目录结构
		例如：tree .	显示当前目录的文件目录结构
		          tree ~	显示当前用户目录的文件目录结构
		          tree -d .	只显示当前目录的目录结构（不显示文件）
	cp：
		cp 源文件 目标文件    （复制文件或目录）	（ps：cp ~/xxx .	表示将用户目录下的xxx复制到当前目录，不改变xxx的名称
						          cp ~/xxx ./yyy		表示将用户目录下的xxx复制到当前目录，并重命名为yyy
		-i：覆盖文件前提示
		-r：若给出的源文件是目录文件，则cp将复制该目录下的所有子目录和文件，目标文件必须是个目录名
	mv：	mv 源文件 目标文件    （移动文件或目录）
		mv ./readme.txt ./demo.txt 	将源文件移动到目标文件并重命名


#其他
	echo：
		会在终端中显示参数指定的文字，通常会和重定向联合使用
	重定向：>和>>
		>:表示输出，会覆盖文件原有的内容
		>>:表示追加，会将内容追加到已有文件的末尾

#远程管理命令
	shutdown：
		shutdown -r	重新启动
		tips：不指定选项和参数，默认1分钟之后 关闭电脑
		        远程维护服务器时，最好不要关闭系统，而应该重新启动系统

		shutdown -r now 	重新启动系统，now表示现在
		shutdown now	立刻关机
		shutdown 20.25 	系统在今天的20.25会关机
		shutdown -c 	取消之前之前的关机计划

	ifconfig:	
		ifconfig	configure a network interface 查看/配置计算机当前的网卡配置信息
	ping:
		ping ip地址	ping		检测到目标ip地址的连接是否正常
